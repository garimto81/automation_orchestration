# .NET Ticks → Python datetime 변환 from datetime import datetime, timezone, timedelta def ticks_to_datetime(ticks: int) -> datetime: """Convert .NET ticks to Python datetime""" epoch = datetime(1, 1, 1, tzinfo=timezone.utc) return epoch + timedelta(microseconds=ticks // 10)

import re def parse_duration(duration_str: str) -> float: """ISO 8601 Duration → 초 단위""" pattern = r"PT(?:(\d+)H)?(?:(\d+)M)?(?:([\d.]+)S)?" match = re.match(pattern, duration_str) if not match: return 0.0 hours = int(match.group(1) or 0) minutes = int(match.group(2) or 0) seconds = float(match.group(3) or 0) return hours * 3600 + minutes * 60 + seconds
